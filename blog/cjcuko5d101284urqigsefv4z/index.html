前端架构师的YY定义

发表于 2010-12-09

 


var 主题 = “The Time is Now for Front-End Architects”;

var 前端架构师 = {
    职责:{
        1: “提升网站的前端性能，保证前端应用具有跨浏览器和跨平台兼容性及可访问性”,
        2: “保证前端团队的编码质量，提升其工作效率”,
        3: “创建并完善内部所使用的前端工具和框架”,
        4: “定义与后端工程师合作的边界、标准化与后端程序的接口”
    },

    技能: {
        1: “XHTML”,
        2: “CSS(1, 2, 3)”,
        3: “跨浏览器和跨平台兼容性处理”,
        4: “DOM脚本编程”,
        5: “AJAX”,
        6: “Flash”,
        7: “渐进增强和适度降级（Progressive Enhancement and Graceful Degradation）”,
        8: “通用性（Accessibility）”,
        9: “可用性”,
        10: “信息架构”,
        11: “界面设计”,
        12: “视觉设计”,
        13: “表现层逻辑（JSP, ASPX等）”,
        14: “商业规则和逻辑”
    }
}

switch (您眼中的前端架构师) {
    case 软件工程师:
        alert(“冷，回归 C/S, 您才更健康。”);
        break;
    case 美工:
        alert(“汗，一个网页不等于一张图片。”);
        break;
    case 用户体验设计师:
        alert(“呃，小伙编程经验不够嘛。”);
        break;
    case 前端工程师:
        alert(“不错，还需付出更多的努力。”);
        break;
    case 前端架构师:
        alert(“恭喜您跟上了 Web 时代！”);
        break;
    default:
        alert(“杯具将要诞生！”);
        break;
}
